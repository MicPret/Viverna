cmake_minimum_required(VERSION 3.24.0 FATAL_ERROR)

if(CMAKE_SYSTEM_NAME STREQUAL Android)
    set(VIVERNA_GAME_PLATFORM_PATH "${CMAKE_CURRENT_SOURCE_DIR}/android")
elseif(CMAKE_SYSTEM_NAME STREQUAL Windows OR CMAKE_SYSTEM_NAME STREQUAL Linux)
    set(VIVERNA_GAME_PLATFORM_PATH "${CMAKE_CURRENT_SOURCE_DIR}/desktop")
endif()

target_sources(${PROJECT_NAME} PRIVATE
    "${VIVERNA_GAME_PLATFORM_PATH}/main.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/Application.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/CameraController.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/Fruit.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/Snake.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/Walls.cpp"
)
